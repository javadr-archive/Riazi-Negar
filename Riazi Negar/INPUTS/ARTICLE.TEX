        \refstepcounter{part}
        \addcontentsline{toc}{part}{\thepart
        \hspace{1em}#1}\else
      \addcontentsline{toc}{part}{#1}\fi
   {\parindent \z@ \raggedleft    %%MAM
    \interlinepenalty \@M
    \ifnum \c@secnumdepth >\m@ne
    \Large \nasb \partname~\beginL\thepart\endL % THEN Print '\partname' %%MAM
      \par\nobreak
    \fi
    \huge \nasb  %%MAM
    #2%
    \markboth{}{}\par
   }\nobreak
   \vskip 3ex
   \@afterheading
  }
